# Tabs

# 基本用法

<template>
    <u-tabs v-model="tabValue">
        <u-tab title="tab1">测试1tab内容</u-tab>
        <u-tab title="tab2">测试2tab内容</u-tab>
        <u-tab title="tab3">测试3tab内容</u-tab>
    </u-tabs>
</template>

<script>
export default {
    data() {
        return { tabValue: 1 }
    }
}
</script>

# to

<template>
    <u-tabs :value="2">
        <u-tab title="去往首页" to="/"></u-tab>
        <u-tab title="去往Demo" to="/demo"></u-tab>
        <u-tab title="去往Tab" to="/tabs"></u-tab>
    </u-tabs>
</template>

# title

    <u-tabs>
        <u-tab title="tab1">

        </u-tab>
        <u-tab title="tab2">
            <span slot="title">

            </span>
            测试1tab内容

</template>

# closable

<template>
    <u-tabs :closable="true">
        <u-tab title="tab1">测试1tab内容</u-tab>
        <u-tab title="tab2">测试2tab内容</u-tab>
        <u-tab title="tab3">测试3tab内容</u-tab>
    </u-tabs>
</template>
