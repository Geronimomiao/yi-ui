  <div>
    <u-button @click="show3 = !show3">Click Me</u-button>

      <u-transition>

          <div class="transition-box">u-transition</div>

        </div>
      </u-transition>
    </div>
  </div>
</template>

    data: () => ({
      show3: true
    })
  }
</script>
